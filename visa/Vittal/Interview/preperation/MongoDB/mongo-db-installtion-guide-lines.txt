#1 Download MONGO DB ZIP file from website with relevant bits 32||64 bits
#2Create Folder in Main drive called MongoDB
C:\MongoDB
#3 Bring ZIP file in this folder and extract all information of ZIP file here
#4 Create folder called data and create sub folders in data folder called db and log
C:\MongoDB\data
C:\MongoDB\data\db
C:\MongoDB\data\log
#5 Configuration 
In C:\MongoDB , create a text file called mongo.config 

	bind_ip = 127.0.0.1
	port = 27017
	quiet=true
	dbpath = C:\MongoDB\data\db
	logpath = C:\MongoDB\data\log\mongodb.log
	logappend = true
	
	copy all these data to config file(please remove the .txt from mangodb.config.txt if you have)

#6 open command prompt as administrator  
Win+X, Press Letter A
#6.a Navigate to C:\MongoDB\mongodb-win32-x86_64-v2.4-latest
#7 Test MongoDB server
>mongod.exe --config="C:\MongoDB\mongo.config"
